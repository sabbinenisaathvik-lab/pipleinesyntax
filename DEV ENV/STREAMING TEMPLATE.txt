INPUT_TYPE={{inputFormat}}
OUTPUT_TYPE={{outputFormat}}

# Validate inputs
if [ -z "$INPUT_TYPE" ] || [ -z "$OUTPUT_TYPE" ]; then
    echo "Usage: $0 <sdi|srt> <srt|multicast>"
    exit 1
fi

if [ "$INPUT_TYPE" == "sdi" ] && [ "$OUTPUT_TYPE" == "srt" ]; then
    echo "Starting SDI to SRT stream..."
    GST_DEBUG_COLOR_MODE=off \
    GST_DEBUG=GST_TRACER:7 \
    GST_TRACERS="latency(flags=pipeline)" \
    GST_DEBUG_FILE={{{logpath}}} \
    /opt/ixgstreaming --i {{inputFormat}} --isdimode {{sdi_mode}} --isdidevice {{dev_no}} \
    --vcodec {{video_codec}} --rcmode {{bitrate_mode}} --vbitrate {{video_bitrate}} \
    --vmaxbitrate {{max_bitrate}} --gopsize {{gop_size}} --preset medium \
    --acodec {{audio_codec}} --abitrate {{audio_bitrate}} --samplerate {{samplerate}} \
    --o {{outputFormat}} --osrtip "{{{output_ip}}}" --osrtport {{output_port}} --osrtmode {{output_mode}} --logpath {{{logpath}}}

elif [ "$INPUT_TYPE" == "sdi" ] && [ "$OUTPUT_TYPE" == "multicast" ]; then
    echo "Starting SDI to Multicast stream..."
    GST_DEBUG_COLOR_MODE=off \
    GST_DEBUG=GST_TRACER:7 \
    GST_TRACERS="latency(flags=pipeline)" \
    GST_DEBUG_FILE={{{logpath}}} \
   /opt/ixgstreaming --i {{inputFormat}} --isdimode {{sdi_mode}} --isdidevice {{dev_no}} \
    --vcodec {{video_codec}} --rcmode {{bitrate_mode}} --vbitrate {{video_bitrate}} \
    --vmaxbitrate {{max_bitrate}} --gopsize {{gop_size}} --preset medium \
    --acodec {{audio_codec}} --abitrate {{audio_bitrate}} --samplerate {{samplerate}} \
    --o {{outputFormat}} --omulticastip "{{{output_ip}}}" --omulticastport {{output_port}} --omulticasttargetdev "{{{target_ip}}}" --logpath {{{logpath}}}

elif [ "$INPUT_TYPE" == "srt" ] && [ "$OUTPUT_TYPE" == "srt" ]; then
    echo "Starting SDI to Multicast stream..."
    GST_DEBUG_COLOR_MODE=off \
    GST_DEBUG=GST_TRACER:7 \
    GST_TRACERS="latency(flags=pipeline)" \
    GST_DEBUG_FILE={{{logpath}}} \
   /opt/ixgstreaming --i {{inputFormat}} --isrtip {{{input_ip}}} --isrtport {{input_port}} --isrtmode {{input_mode}} --isrtlatency {{latency}} \
    --vcodec {{video_codec}} --rcmode {{bitrate_mode}} --vbitrate {{video_bitrate}} \
    --vmaxbitrate {{max_bitrate}} --gopsize {{gop_size}} --preset medium \
    --acodec {{audio_codec}} --abitrate {{audio_bitrate}} --samplerate {{samplerate}} \
    --o {{outputFormat}} --osrtip {{{output_ip}}} --osrtport {{output_port}} --osrtmode {{output_mode}}  --logpath {{{logpath}}}
    
elif [ "$INPUT_TYPE" == "srt" ] && [ "$OUTPUT_TYPE" == "multicast" ]; then
    echo "Starting SDI to Multicast stream..."
    GST_DEBUG_COLOR_MODE=off \
    GST_DEBUG=GST_TRACER:7 \
    GST_TRACERS="latency(flags=pipeline)" \
    GST_DEBUG_FILE={{{logpath}}} \
   /opt/ixgstreaming --i {{inputFormat}} --isrtip {{{input_ip}}} --isrtport {{input_port}} --isrtmode {{input_mode}} --isrtlatency {{latency}} \
    --vcodec {{video_codec}} --rcmode {{bitrate_mode}} --vbitrate {{video_bitrate}} \
    --vmaxbitrate {{max_bitrate}} --gopsize {{gop_size}} --preset medium \
    --acodec {{audio_codec}} --abitrate {{audio_bitrate}} --samplerate {{samplerate}} \
    --o {{outputFormat}} --omulticastip "{{{output_ip}}}" --omulticastport {{output_port}} --omulticasttargetdev "{{{target_ip}}}" --logpath {{{logpath}}}
    
elif [ "$INPUT_TYPE" == "multicast" ] && [ "$OUTPUT_TYPE" == "multicast" ]; then
    echo "Starting SDI to Multicast stream..."
    GST_DEBUG_COLOR_MODE=off \
    GST_DEBUG=GST_TRACER:7 \
    GST_TRACERS="latency(flags=pipeline)" \
    GST_DEBUG_FILE={{{logpath}}} \
   /opt/ixgstreaming --i {{inputFormat}} --imulticastip {{input_ip}} --imulticastport {{input_port}} --imulticasttargetdev {{{network_dev_src}}} \
    --vcodec {{video_codec}} --rcmode {{bitrate_mode}} --vbitrate {{video_bitrate}} \
    --vmaxbitrate {{max_bitrate}} --gopsize {{gop_size}} --preset medium \
    --acodec {{audio_codec}} --abitrate {{audio_bitrate}} --samplerate {{samplerate}} \
    --o {{outputFormat}} --omulticastip "{{{output_ip}}}" --omulticastport {{output_port}} --omulticasttargetdev "{{{target_ip}}}" --logpath {{{logpath}}}

elif [ "$INPUT_TYPE" == "multicast" ] && [ "$OUTPUT_TYPE" == "srt" ]; then
    echo "Starting SDI to Multicast stream..."
    GST_DEBUG_COLOR_MODE=off \
    GST_DEBUG=GST_TRACER:7 \
    GST_TRACERS="latency(flags=pipeline)" \
    GST_DEBUG_FILE={{{logpath}}} \
   /opt/ixgstreaming --i {{inputFormat}} --imulticastip {{input_ip}} --imulticastport {{input_port}}  --imulticasttargetdev {{{network_dev_src}}} \
    --vcodec {{video_codec}} --rcmode {{bitrate_mode}} --vbitrate {{video_bitrate}} \
    --vmaxbitrate {{max_bitrate}} --gopsize {{gop_size}} --preset medium \
    --acodec {{audio_codec}} --abitrate {{audio_bitrate}} --samplerate {{samplerate}} \
    --o {{outputFormat}} --osrtip {{{output_ip}}} --osrtport {{output_port}} --osrtmode {{output_mode}} --logpath {{{logpath}}}
else
    echo "Invalid input or output type. Use 'sdi' for input and either 'srt' or 'multicast' for output."
    exit 1
fi
